Add binary strings:

Given two binary strings, return their sum (also a binary string).

Example:

a = "100"

b = "11"
Return a + b = â€œ111â€.



simple Implementation:

string Solution::addBinary(string a, string b) {
     string str="";int carry=0;int i=a.size()-1;int j=b.size()-1;
      while(i>=0 || j>=0){
          int numa=0;int numb=0;int sum=0;
          if(i>=0)
           numa=(a[i]-'0');
          if(j>=0)
          numb=(b[j]-'0');
          sum=numa+numb;
          sum+=carry;
          if(sum==1){
              sum=1;
              carry=0;
          }
          else if(sum==2)
          {
              sum=0;
              carry=1;
          }
          else if(sum==3){
              sum=1;
              carry=1;
          }
          str=char(sum+'0')+str;
          i--;j--;  
      }
        if(carry)
            str=char(carry+'0')+str;
        return str;
            
}


TC:O(n)
SC:O(1)
